```python
"""Chatlytics tool handlers for Hermes (HERMES-05).

Each tool is a 3-tuple in TOOLS: (name, schema, handler). The plugin
register() (adapter.py) iterates this tuple and calls ctx.register_tool
for each entry.

Handler contract:
    async def chatlytics_xxx(client: ChatlyticsClient, **kwargs) -> dict

Return shape: {"success": bool, ...}.
"""
from __future__ import annotations
import logging
from typing import Any, Dict, Tuple, List, Callable, Awaitable
import httpx
from .client import ChatlyticsClient

logger = logging.getLogger("chatlytics_hermes.tools")
Handler = Callable[..., Awaitable[Dict[str, Any]]]

# --- response helpers ---
def _ok(payload: Any) -> Dict[str, Any]:
    if isinstance(payload, dict):
        out = {"success": True}
        out.update(payload)
        out["success"] = True  # never let payload-side success=False override
        return out
    return {"success": True, "result": payload}

def _err(response: httpx.Response | None, exc: Exception | None = None) -> Dict[str, Any]:
    if exc is not None:
        return {"success": False, "error": f"{type(exc).__name__}: {exc}"}
    assert response is not None
    try:
        payload = response.json()
    except Exception:  # noqa: BLE001
        payload = {"raw_text": response.text}
    msg = (payload.get("error") if isinstance(payload, dict) else None) or f"HTTP {response.status_code}"
    return {"success": False, "error": msg, "status_code": response.status_code, "raw_response": payload}

async def _post(client, path, body) -> Dict[str, Any]:
    try:
        r = await client.post(path, json=body)
    except httpx.RequestError as exc:
        return _err(None, exc)
    if r.status_code >= 400:
        return _err(r)
    try:
        return _ok(r.json())
    except Exception:  # noqa: BLE001
        return _ok({"raw_text": r.text})

async def _get(client, path, params=None) -> Dict[str, Any]:
    try:
        r = await client.get(path, params=params)
    except httpx.RequestError as exc:
        return _err(None, exc)
    if r.status_code >= 400:
        return _err(r)
    try:
        return _ok(r.json())
    except Exception:  # noqa: BLE001
        return _ok({"raw_text": r.text})
```

2. In `register(ctx)`, after `ctx.register_platform(...)`, add:
```python
from .tools import TOOLS

def _make_handler(handler):
    async def _bound(**kwargs):
        # Resolve adapter/client at call time -- register() runs before
        # connect(), so the adapter instance may not yet have a live client.
        try:
            entry = ctx.get_platform("chatlytics")
        except (AttributeError, KeyError, Exception):  # noqa: BLE001
            entry = None
        adapter = getattr(entry, "adapter", None) if entry else None
        client = getattr(adapter, "client", None) if adapter else None
        if client is None:
            return {"success": False, "error": "Chatlytics adapter not connected; call adapter.connect() before invoking tools."}
        # Media handlers also need the adapter (for send_media uploads).
        # Use Python's inspect to detect 'adapter' parameter.
        import inspect
        sig = inspect.signature(handler)
        if "adapter" in sig.parameters:
            return await handler(client, adapter=adapter, **kwargs)
        return await handler(client, **kwargs)
    return _bound

if hasattr(ctx, "register_tool"):
    for name, schema, handler in TOOLS:
        ctx.register_tool(
            name=name,
            toolset="chatlytics",
            schema=schema,
            handler=_make_handler(handler),
        )
```

The `hasattr(ctx, "register_tool")` guard keeps the HERMES-01 MockCtx (which only implements `register_platform`) green; the existing tests do not need to know about tool registration. The new `test_tools.py` / `test_tool_schemas.py` test suites either (a) test handlers directly bypassing register, or (b) use a richer MockCtx.
