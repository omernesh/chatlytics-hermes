The contract documented in the phase brief takes precedence over the base shape for this plugin — long-running tool handlers (HERMES-05) will use the context-manager form.

def _env_enablement() -> dict | None:
    home = os.getenv("CHATLYTICS_HOME_CHANNEL", "").strip()
    if not home:
        return None
    return {"home_channel": {"chat_id": home, "name": home}}

async def _standalone_send(text: str, **kwargs) -> dict:
    base_url = os.getenv("CHATLYTICS_BASE_URL", "").strip()
    api_key = os.getenv("CHATLYTICS_API_KEY", "").strip()
    home_channel = os.getenv("CHATLYTICS_HOME_CHANNEL", "").strip()
    if not (base_url and api_key and home_channel):
        return {"error": "Chatlytics standalone send: BASE_URL, API_KEY and HOME_CHANNEL must be configured"}
    async with httpx.AsyncClient(base_url=base_url.rstrip("/"), timeout=30.0, headers={"Authorization": f"Bearer {api_key}", "User-Agent": USER_AGENT}) as client:
        response = await client.post("/api/v1/send", json={"chatId": home_channel, "text": text})
        try:
            payload = response.json()
        except Exception:
            payload = {"raw_text": response.text}
        if response.status_code == 200 and isinstance(payload, dict) and payload.get("success", True):
            return {"success": True, **payload}
        return {"success": False, "error": payload.get("error") if isinstance(payload, dict) else f"HTTP {response.status_code}", "raw_response": payload}
```

async def post_multipart(
    self,
    path: str,
    *,
    files: Dict[str, tuple],   # {"file": (filename, bytes, content_type)}
    data: Optional[Dict[str, Any]] = None,
) -> httpx.Response:
    return await self._client.post(path, files=files, data=data or {})

async def send_media(self, payload: Dict[str, Any]) -> httpx.Response:
    """POST /api/v1/send-media with the given JSON payload."""
    return await self._client.post("/api/v1/send-media", json=payload)

async def upload_file(
    self,
    filename: str,
    content: bytes,
    content_type: str = "application/octet-stream",
) -> httpx.Response:
    """POST /api/v1/upload as multipart/form-data; returns response with {url: ...}."""
    return await self.post_multipart(
        "/api/v1/upload",
        files={"file": (filename, content, content_type)},
    )
```

Why both `send_media` and `post_multipart`: keeping the multipart primitive separate makes future helpers (e.g. signed-upload buckets) reuse the same low-level wrapper without duplicating the form-encoding logic.

For each handler the shape is identical:
1. Guard `self._client is None` → return `SendResult(success=False, error="Adapter not connected: ...")`.
2. Resolve the resource → `media_url: str`:
   - If `isinstance(resource, (bytes, bytearray))` → upload, get URL.
   - Elif `isinstance(resource, str) and resource.startswith(("http://", "https://"))` → use as-is.
   - Elif `isinstance(resource, str)` → treat as local path: `with open(resource, "rb") as fh: content = fh.read()` → upload, get URL. For `send_image_file` this is the only branch.
3. Build payload: `{"chatId": chat_id, "mediaType": ..., "mediaUrl": media_url}`. Optional: `caption`, `filename` (for documents).
4. POST via `self._client.send_media(payload)`; convert response to `SendResult` using the same success/error logic as `send()`.

Document handler additionally passes `filename` to the upload + payload (use the argument `file_name` per base signature, but also accept positional `filename` via `**kwargs` for caller convenience).

`_keep_typing`:

```python
@contextlib.asynccontextmanager
async def _keep_typing(self, chat_id: str, interval: float = 30.0):
    """30 s typing-bubble heartbeat for long-running tool handlers (HERMES-04).

    Fires `send_typing(chat_id, duration=30.0)` immediately, then every
    `interval` seconds until the context manager exits.  Cancels the
    background task cleanly on exit (even if the body raises).
    """
    async def _beat() -> None:
        try:
            while True:
                try:
                    await self.send_typing(chat_id, duration=30.0)
                except Exception:  # noqa: BLE001 — heartbeat must never propagate
                    logger.debug("send_typing heartbeat raised; continuing", exc_info=True)
                await asyncio.sleep(interval)
        except asyncio.CancelledError:
            pass

    # Fire one immediately so the bubble appears without waiting `interval`.
    try:
        await self.send_typing(chat_id, duration=30.0)
    except Exception:  # noqa: BLE001
        logger.debug("send_typing initial fire raised; continuing", exc_info=True)

    task = asyncio.create_task(_beat())
    try:
        yield
    finally:
        task.cancel()
        try:
            await task
        except (asyncio.CancelledError, Exception):  # noqa: BLE001 — never leak teardown errors
            pass
```

Add `import asyncio` and `import contextlib` to the adapter module.
